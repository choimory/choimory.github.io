# 설정

## TypeORM 의존성 추가

```bash
# typeorm, nestjs/typeorm 모듈
npm install --save typeorm @nestjs/typeorm

# postgres 모듈
npm install --save pg

# mariadb, mysql 모듈
npm install --save mysql2

# 한번에
npm install --save typeorm @nestjs/typeorm mysql2
npm install --save typeorm @nestjs/typeorm pg
```

- @nestjs/typeorm: Nest.js의 TypeORM 연결 모듈
- typeorm: TypeORM 모듈
- pg: PostgreSQL 모듈
- mysql2: MySQL, MariaDB 모듈

# AppModule에 TypeORM 설정 작성

```tsx
@Module({
  imports: [
    MemberModule,
    ConfigModule.forRoot({
      // 실행할 .env 파일을 지정
      // 실행 명령어로 전달받은 NODE_ENV 값으로 파일명을 구분하여 해당 실행환경 로드
      envFilePath: process.env.NODE_ENV
        ? `.env.${process.env.NODE_ENV}`
        : '.env',

      // ConfigModule 전역 설정
      isGlobal: true,

      // Joi를 이용해 .env 값들을 검증
      validationSchema: Joi.object({
        NODE_ENV: Joi.string().valid('local', 'dev', 'prod').required(),
        PORT: Joi.number().port().required(),
        DB_TYPE: Joi.string().required(),
        DB_HOST: Joi.string().required(),
        DB_PORT: Joi.number().required(),
        DB_USERNAME: Joi.string().required(),
        DB_PASSWORD: Joi.string().required(),
        DB_DATABASE: Joi.string().required(),
      }),
    } as ConfigModuleOptions),

    // TypeORM 설정
    TypeOrmModule.forRoot({
      type: process.env.DB_TYPE,
      host: process.env.DB_HOST,
      port: Number(process.env.DB_PORT),
      username: process.env.DB_USERNAME,
      password: process.env.DB_PASSWORD,
      database: process.env.DB_DATABASE,
      schema: process.env.DB_SCHEMA,
      synchronize: process.env.NODE_ENV === 'local',
    } as TypeOrmModuleOptions),
  ],
  controllers: [AppController],
  providers: [AppService],
})
export class AppModule {}

```

- TypeOrmModule.forRoot({}) 추가하는걸로 Repository, DataSource등을 사용할 준비가 모두 완료된다

# Entity 작성

```tsx
import {
  BeforeInsert,
  Column,
  Entity,
  OneToMany,
  PrimaryColumn,
} from 'typeorm';
import { CommonTime } from '../../common/entities/common-time.entity';
import { v7 as uuid } from 'uuid';
import { MemberSuspension } from './member-suspension.entity';

@Entity()
export class Member extends CommonTime {
  @PrimaryColumn('uuid')
  id: string;

  @Column({ unique: true, nullable: false })
  email: string;

  @Column({ unique: true, nullable: false })
  nickname: string;

  @Column({ nullable: false })
  password: string;

  @Column({ nullable: true })
  image: string;

  @Column({ nullable: true })
  introduce: string;

  @OneToMany(
    () => MemberSuspension,
    (memberSuspension) => memberSuspension.member,
  )
  memberSuspension: MemberSuspension;

  constructor(
    createdAt: Date,
    modifiedAt: Date,
    deletedAt: Date,
    id: string,
    email: string,
    nickname: string,
    password: string,
    image: string,
    introduce: string,
    memberSuspension: MemberSuspension,
  ) {
    super(createdAt, modifiedAt, deletedAt);
    this.id = id;
    this.email = email;
    this.nickname = nickname;
    this.password = password;
    this.image = image;
    this.introduce = introduce;
    this.memberSuspension = memberSuspension;
  }

  @BeforeInsert()
  beforeInsert() {
    this.id = uuid();
  }
}

```

- 엔티티

```tsx
import { Module } from '@nestjs/common';
import { MemberService } from './member.service';
import { MemberController } from './member.controller';
import { TypeOrmModule } from '@nestjs/typeorm';
import { Member } from './entities/member.entity';
import { MemberSuspension } from './entities/member-suspension.entity';

@Module({
  controllers: [MemberController],
  providers: [MemberService],
  imports: [TypeOrmModule.forFeature([Member, MemberSuspension])], //모듈에 엔티티 등록
})
export class MemberModule {}

```

- 작성한 엔티티는, 엔티티가 해당하는 도메인의 모듈에서 imports 해준다
    - 엔티티용 별도의 디렉토리를 만들거나, 와일드카드를 이용해서 엔티티들을 일괄 등록하는 방법도 있음
    - 모듈을 이해하고, 도메인별 모듈 관리를 깔끔하게 나누어 하기 위해 이런식으로 진행해봤음

# Repository 사용

## 기본 추상화된 API 사용

```tsx
import { HttpStatus, Injectable } from '@nestjs/common';
import { DataSource, QueryRunner, Repository } from 'typeorm';
import { Member } from './entities/member.entity';
import { InjectRepository } from '@nestjs/typeorm';
import { CommonResponseDto } from '../common/dto/response/common.response.dto';

@Injectable()
export class MemberService {
  constructor(
    @InjectRepository(Member)
    private readonly memberRepository: Repository<Member>, //Repository 주입
    private readonly dataSource: DataSource, //TypeOrmModule.forRoot()설정했으면 DataSource도 별도 작업없이 주입받아 사용가능
  ) {}

  async find(id: string) {
    // repository api
    const response = await this.memberRepository.find({ where: { id: id } });

    return new CommonResponseDto(
      HttpStatus.OK,
      HttpStatus[HttpStatus.OK],
      response,
    );
  }
}

```

- 별도의 리포지토리 작성없이, 서비스 레이어에서 특정 엔티티의 Repository를 바로 주입받아 사용할 수 있다

# 상세 쿼리

```tsx
import { HttpStatus, Injectable } from '@nestjs/common';
import { DataSource, QueryRunner, Repository } from 'typeorm';
import { Member } from './entities/member.entity';
import { InjectRepository } from '@nestjs/typeorm';
import { CommonResponseDto } from '../common/dto/response/common.response.dto';

@Injectable()
export class MemberService {
  constructor(
    @InjectRepository(Member)
    private readonly memberRepository: Repository<Member>,
    private readonly dataSource: DataSource,
  ) {}

  async find(id: string) {
    // query building
    const response = await this.memberRepository
      .createQueryBuilder('m')
      .where('m.id=:id', { id: id })
      .andWhere('m.nickname=:nickname', { nickname: 'abc' })
      .getOne();

    return new CommonResponseDto(
      HttpStatus.OK,
      HttpStatus[HttpStatus.OK],
      response,
    );
  }
}

```

- 주입받은 Repository에서 QueryBuilder를 생성해 상세 쿼리를 작성할 수 있다

- 간결한 트랜잭션 문법

```tsx
import { HttpStatus, Injectable } from '@nestjs/common';
import { DataSource, QueryRunner, Repository } from 'typeorm';
import { Member } from './entities/member.entity';
import { InjectRepository } from '@nestjs/typeorm';
import { CommonResponseDto } from '../common/dto/response/common.response.dto';

@Injectable()
export class MemberService {
  constructor(
    @InjectRepository(Member)
    private readonly memberRepository: Repository<Member>,
    private readonly dataSource: DataSource,
  ) {}

  async find(id: string) {
    // create query runner
    const queryRunner: QueryRunner = this.dataSource.createQueryRunner();

    //start transaction
    await queryRunner.startTransaction();

    try {
      await queryRunner.manager.save(
        new Member(
          new Date(),
          new Date(),
          new Date(),
          'null',
          'null',
          'null',
          'null',
          'null',
          'null',
          undefined,
        ),
      );

      await queryRunner.commitTransaction();
    } catch (e) {
      await queryRunner.rollbackTransaction();
    } finally {
      await queryRunner.release();
    }

    return new CommonResponseDto(
      HttpStatus.OK,
      HttpStatus[HttpStatus.OK],
      null,
    );
  }
}

```

- 상세한 트랜잭션 제어
1. DataSource에서 QueryRunner생성
2. 만든 QueryRunner로 트랜잭션 시작
3. try catch finally로 commit, rollback, release 분기
